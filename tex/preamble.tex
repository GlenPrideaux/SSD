% --- SSDPart environment (format defined in preamble) ---
\newenvironment{SSDPart}[1]{%
  % BEGIN code
  \cleardoublepage            % start on a fresh page (book-like)
  \thispagestyle{plain}       % or empty
  \setcounter{footnote}{0}    % reset footnotes (optional)
  \vspace{10pt}\Needspace{10\baselineskip}\begin{center}\small The word concerning\vspace{-10pt}
    \section*{#1}
  \end{center}
  \nobreak\nointerlineskip\penalty10000
  \begin{multicols}{2}
}{%
  % END code
  \par\medskip
  \end{multicols}
}
